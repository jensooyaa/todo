## 二、注册流程

### 2.1 流程图

```
用户输入密码 "123456"
        ↓
bcrypt.hash("123456", 10)
        ↓
生成哈希值 "$2a$10$xK3f..."（不可逆，无法反推出原始密码）
        ↓
存入数据库 INSERT INTO users (username, password) VALUES ('zhang', '$2a$10$xK3f...')
        ↓
返回 { message: '注册成功', userId: 1 }
```

### 2.2 前端代码

```jsx
// client/src/pages/Login/index.jsx
const handleSubmit = async (e) => {
  e.preventDefault();
  try {
    // 发送注册请求，密码以明文发送（靠 HTTPS 保证传输安全）
    await register({ username, password });
    setMessage('注册成功，请登录');
  } catch (error) {
    setMessage(error.response?.data?.message || '操作失败');
  }
};
```

```js
// client/src/api/user.js
export const register = (data) => request.post('/users/register', data);
// 发送 POST 请求：{ username: 'zhang', password: '123456' }
```

### 2.3 后端代码

```js
// server/controllers/user.js
const bcrypt = require('bcryptjs');
const SALT_ROUNDS = 10;

register: async (req, res) => {
  const { username, password } = req.body;

  // 1. 校验参数
  if (!username || !password) {
    return res.status(400).json({ message: '用户名和密码不能为空' });
  }

  // 2. 检查用户名是否已存在
  const existingUser = await User.findByUsername(username);
  if (existingUser) {
    return res.status(400).json({ message: '用户名已存在' });
  }

  // 3. 对密码进行 bcrypt 哈希加密
  //    SALT_ROUNDS = 10 表示 2^10 = 1024 次迭代，越大越安全但越慢
  //    bcrypt 会自动生成随机 salt，相同密码每次哈希结果都不同
  const hashedPassword = await bcrypt.hash(password, SALT_ROUNDS);
  // hashedPassword 示例: "$2a$10$N9qo8uLOickgx2ZMRZoMyeIjZAgcfl7p92ldGxad68LJZdL17lhWy"
  //                       ^^^^ ^^ ^^^^^^^^^^^^^^^^^^^^^^ ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
  //                       算法  轮数     salt（22位）            哈希值（31位）

  // 4. 存入数据库（存的是哈希值，不是原始密码）
  const userId = await User.create(username, hashedPassword);
  res.status(201).json({ message: '注册成功', userId });
}
```

```js
// server/models/user.js
create: async (username, password) => {
  const [result] = await pool.query(
    'INSERT INTO users (username, password) VALUES (?, ?)',
    [username, password]  // 这里的 password 已经是哈希值了
  );
  return result.insertId;
}
```

### 2.4 数据库中的样子

```
+----+----------+--------------------------------------------------------------+
| id | username | password                                                     |
+----+----------+--------------------------------------------------------------+
|  1 | zhang    | $2b$10$mv2y9aiejaZWiZfL8oHxTu.8Vhyy4J7Fj89o8GoACx7CviFuG9R9K |
|  2 | li       | $2b$10$Xk3fGh8dJf9sLpW2qRvTeOYz5k8mN3pQ1wE4rT6yU8iO2pA3bC5de |
+----+----------+--------------------------------------------------------------+
即使两人密码都是 "123456"，哈希值也完全不同（因为 salt 不同）
```

## 三、登录流程

### 3.1 流程图

```
用户输入密码 "123456"
        ↓
发送 POST /api/users/login { username: 'zhang', password: '123456' }
        ↓
后端查数据库：SELECT * FROM users WHERE username = 'zhang'
        ↓
取出哈希值 "$2b$10$mv2y9aie..."
        ↓
bcrypt.compare("123456", "$2b$10$mv2y9aie...") → true，密码正确
        ↓
jwt.sign({ userId: 1, username: 'zhang' }, JWT_SECRET, { expiresIn: '1d' })
        ↓
生成 token: "eyJhbGciOiJIUzI1NiJ9.eyJ1c2VySWQiOjF9.XXXXXXXXX"
        ↓
res.cookie('JWT', token, { httpOnly: true }) → 写入浏览器 cookie
        ↓
返回 { message: '登录成功', userId: 1, username: 'zhang' }
（注意：token 不在返回体里，在 Set-Cookie 响应头里）
```

### 3.2 前端代码

```jsx
// client/src/pages/Login/index.jsx
const handleSubmit = async (e) => {
  e.preventDefault();
  try {
    const res = await login({ username, password });
    // 只存用户基本信息，token 由浏览器 cookie 自动管理
    localStorage.setItem('user', JSON.stringify(res.data));
    navigate('/');
  } catch (error) {
    setMessage(error.response?.data?.message || '操作失败');
  }
};
```

### 3.3 后端代码

```js
// server/controllers/user.js
const jwt = require('jsonwebtoken');

login: async (req, res) => {
  const { username, password } = req.body;

  // 1. 查找用户
  const user = await User.findByUsername(username);
  if (!user) {
    return res.status(400).json({ message: '用户不存在' });
  }

  // 2. 验证密码
  //    bcrypt.compare 会从哈希值中提取 salt，用同样的 salt 对输入密码哈希，然后比对
  //    不需要自己存 salt，bcrypt 把 salt 嵌在了哈希值里
  const isMatch = await bcrypt.compare(password, user.password);
  if (!isMatch) {
    return res.status(400).json({ message: '用户名或密码错误' });
  }

  // 3. 生成 JWT token
  //    把 userId 和 username 编码进 token
  //    用 JWT_SECRET 签名，防止被篡改
  //    设置过期时间
  const token = jwt.sign(
    { userId: user.id, username: user.username },  // payload：要编码的数据
    process.env.JWT_SECRET,                         // 密钥：用来签名
    { expiresIn: process.env.JWT_EXPIRES_IN || '7d' } // 过期时间
  );

  // 4. 将 token 写入 httpOnly cookie
  res.cookie('JWT', token, {
    httpOnly: true,   // 前端 JS 无法通过 document.cookie 读取，防 XSS 窃取
    maxAge: Number(process.env.COOKIE_MAX_AGE) || 86400000, // 过期时间（毫秒）
    sameSite: 'lax',  // 防 CSRF：只有同站请求或顶级导航才会带上 cookie
  });

  // 5. 返回用户信息（不包含 token，token 在 cookie 里）
  res.json({ message: '登录成功', userId: user.id, username: user.username });
}
```

### 3.4 JWT token 结构

```
eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJ1c2VySWQiOjEsInVzZXJuYW1lIjoiemhhbmciLCJpYXQiOjE3MTAwMDAwMDAsImV4cCI6MTcxMDA4NjQwMH0.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c
|_____________header______________|.|______________________payload______________________|.|_______________signature_______________|

header  = { "alg": "HS256", "typ": "JWT" }           ← 算法信息
payload = { "userId": 1, "username": "zhang",
            "iat": 1710000000, "exp": 1710086400 }    ← 数据 + 签发时间 + 过期时间
signature = HMAC-SHA256(header + payload, JWT_SECRET)  ← 签名（防篡改）
```

- header 和 payload 是 Base64 编码，**任何人都能解码看到内容**
- 但没有 JWT_SECRET 就**无法伪造签名**，篡改 payload 后签名对不上，验证失败

## 四、请求验证流程

### 4.1 流程图

```
浏览器发起请求 GET /api/todos?userId=1
        ↓
浏览器自动带上 cookie: JWT=eyJhbGciOiJI...
        ↓
Express 收到请求
        ↓
cookieParser() 中间件解析 cookie → req.cookies.JWT = "eyJhbGciOiJI..."
        ↓
auth 中间件验证 token
        ↓
    ┌── 验证通过 → req.user = { userId: 1, username: 'zhang' } → 继续执行 controller
    └── 验证失败 → 返回 401 { message: '登录已过期' } → 前端跳转登录页
```

### 4.2 auth 中间件代码

```js
// server/middleware/auth.js
const jwt = require('jsonwebtoken');

const auth = (req, res, next) => {
  // 1. 从 cookie 中取出 token
  const token = req.cookies.JWT;
  if (!token) {
    return res.status(401).json({ message: '未登录，请先登录' });
  }

  try {
    // 2. 验证 token
    //    jwt.verify 会做三件事：
    //    a. 检查签名是否合法（用 JWT_SECRET 重新计算签名，和 token 里的签名对比）
    //    b. 检查是否过期（对比 exp 和当前时间）
    //    c. 解码 payload，返回 { userId, username, iat, exp }
    const decoded = jwt.verify(token, process.env.JWT_SECRET);

    // 3. 将用户信息挂到 req 上，后续 controller 可以直接使用
    req.user = decoded; // { userId: 1, username: 'zhang', iat: ..., exp: ... }
    next(); // 放行，继续执行下一个中间件或 controller
  } catch (error) {
    // token 无效或过期
    return res.status(401).json({ message: '登录已过期，请重新登录' });
  }
};

module.exports = auth;
```

### 4.3 路由中使用 auth 中间件

```js
// server/app.js
const auth = require('./middleware/auth');

app.use('/api/users', userRoutes);                // 登录注册 → 不需要验证
app.use('/api/categories', auth, categoryRoutes); // 分类接口 → 需要登录
app.use('/api/todos', auth, todoRoutes);          // 任务接口 → 需要登录
app.use('/api', auth, subtaskRoutes);             // 子任务   → 需要登录
app.use('/api/comments', auth, commentRoutes);    // 评论接口 → 需要登录

// 请求到达 todoRoutes 之前，会先经过 auth 中间件
// 如果 auth 返回 401，请求就不会到达 todoController
```

### 4.4 前端响应拦截器

```js
// client/src/api/request.js
import axios from 'axios';

const request = axios.create({
  baseURL: '/api',
  timeout: 5000,
});

// 响应拦截器：统一处理 401
request.interceptors.response.use(
  (response) => response,  // 正常响应直接返回
  (error) => {
    if (error.response?.status === 401) {
      // token 过期或无效 → 清除本地用户信息 → 跳转登录页
      localStorage.removeItem('user');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default request;
```

## 五、Cookie 配置详解

```js
res.cookie('JWT', token, {
  httpOnly: true,
  maxAge: 86400000,
  sameSite: 'lax',
});
```

| 属性 | 值 | 作用 |
|------|-----|------|
| `httpOnly` | `true` | 浏览器 JS 无法通过 `document.cookie` 读取此 cookie，防止 XSS 攻击窃取 token |
| `maxAge` | `86400000` | cookie 过期时间（毫秒），86400000 = 1天，应与 JWT expiresIn 保持一致 |
| `sameSite` | `'lax'` | 跨站请求不带 cookie（防 CSRF），同站请求和顶级导航（链接跳转）会带 |

### 为什么 maxAge 要和 JWT expiresIn 一致？

```
情况一：cookie 先过期，JWT 还有效
→ cookie 被浏览器删除，请求不带 token，用户被踢出（JWT 白白浪费）

情况二：JWT 先过期，cookie 还在
→ 浏览器带着过期 token 发请求，每次都 401（cookie 赖着不走）

正确做法：两者设为一样（都是 1 天）
→ JWT 过期的同时 cookie 也被清除
```

## 七、安全分工总结

```
用户输入密码 "123456"
    │
    │  ① 传输层安全 —— HTTPS（TLS加密）
    │     防止中间人窃听，即使截获请求也看不到内容
    │
    ▼
后端收到密码
    │
    │  ② 存储层安全 —— bcrypt 哈希
    │     数据库只存哈希值，拖库后无法反推原始密码
    │     暴力破解极慢（每次 ~100ms），彩虹表无效（每个密码 salt 不同）
    │
    ▼
验证通过，签发 JWT
    │
    │  ③ 身份验证 —— JWT 签名
    │     用 JWT_SECRET 签名，没有密钥无法伪造 token
    │     设置过期时间，限制 token 有效期
    │
    ▼
写入 httpOnly cookie
    │
    │  ④ 防 XSS —— httpOnly: true
    │     前端 JS 无法读取 cookie，恶意脚本偷不走 token
    │
    │  ⑤ 防 CSRF —— sameSite: 'lax'
    │     跨站请求不带 cookie，其他网站无法冒用你的身份
    │
    ▼
后续请求自动携带 cookie → auth 中间件验证 → 放行或拒绝
```

| 环节 | 方案 | 防什么攻击 |
|------|------|-----------|
| **传输** | HTTPS（TLS） | 中间人窃听（MITM） |
| **存储** | bcrypt 哈希（10轮 salt） | 数据库泄露后暴力破解密码 |
| **身份验证** | JWT 签名 | 伪造身份、篡改用户信息 |
| **token 存储** | httpOnly cookie | XSS 攻击窃取 token |
| **跨站防护** | sameSite: lax | CSRF 跨站请求伪造 |
| **过期控制** | expiresIn + maxAge | 限制 token 生命周期 |
